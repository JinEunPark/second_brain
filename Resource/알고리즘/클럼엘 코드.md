package run;  
  
import java.io.*;  
import java.util.*;  
import java.util.stream.Collectors;  
  
public class Main {  
  
    public static void main(String[] args) throws IOException {  
  
        BufferedReader br = new BufferedReader(new FileReader("/Users/bagjin-eun/Downloads/Q1_Input.txt"));  
        StringBuilder sb = new StringBuilder();  
        String t = "";  
        while((t = br.readLine()) != null ){  
            String[] tmp  = t.split("");  
            int l = sb.length();  
            for(int i = 0; i< tmp.length/2; i++) {  
                int e = tmp.length - i -1;  
                if (!tmp[e].equals(tmp[i])) {  
                    sb.append(false).append("\n");  
                    break;                }  
            }  
  
            if(l != sb.length()) continue;  
            sb.append(true).append("\n");  
  
        }  
  
        System.out.println(sb);  
  
  
  
        return;    
        }  
}

package run;  
  
import java.io.*;  
import java.util.*;  
import java.util.stream.Collectors;  
  
public class Main {  
  
    public static void main(String[] args) throws IOException {  
  
        BufferedReader br = new BufferedReader(new FileReader("/Users/bagjin-eun/Downloads/http.log.sample"));  
        StringBuilder sb = new StringBuilder();  
  
        String t = "";  
        while((t = br.readLine()) != null ){  
            if(t.charAt(0) == '#') continue;  
            t = t.replaceAll(","," ");  
            t = t.replaceAll("\t", ",");  
            sb.append(t);  
        }  
        File file = new File("/Users/bagjin-eun/Downloads/http.log.sample2");  
        BufferedWriter bw = new BufferedWriter(new FileWriter(file));  
        bw.write(sb.toString());  
        bw.close();
        return;    
        }  
}
